	}

        // Do nothing.
    }

     * Do nothing.
     * @param stats the extract stats.

   public void setStats(final ExtractStats stats)
    {
        // Do nothing.

    }

     * Do nothing.
     */
    @Override
   public void run()
    {
        // Do nothing.
    }
}
